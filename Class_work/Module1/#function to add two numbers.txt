#function to add two numbers
def addNumber(a, b):
    return a + b

#function to calculate the difference of two numbers
def calculateDifference(a, b):
    return a - b

#function to calculate the product of two numbers
def calculateProduct(a, b):
    return a * b

#function to calculate the division
def calculateDivision(a, b):
    return a / b 

#fuction to calculate the modulus
def calculateModulus(a, b):
    return a % b
